  PKU 3617 Best Cow Line和PKU 3623 Best Cow Line 解题报告 收藏

PKU 3617和PKU 3623是一样的题，只是数据范围不同。
3617是2000个字符，而3623是30000个字符。只是3623的时限比较松，贪心也可以过。

本题有两种解法，分别是贪心O(n^2)和后缀数组O(nlogn)

后缀数组法：

  1. /*
  2.   Title: Best Cow Line
  3.   Author: Jeff
  4.   Time:  2008/10/30
  5.   Complexity: O(Nlog(N));
  6.   Result: 2256K 766MS/ AC
  7.   Reference: 
  8.         
  9.   Description: 
  10.         给定一个字符数组，可以从数组的开头或者结尾取出元素，
  11.         按取出顺序排成一列，使得他的字典序最小。
  12.   Tips:
  13.         后缀数组 
  14.         将原字符数组和字符数组的逆序合在一次进行后缀数组排序，
  15.         然后对rank数组前N和后N个数进行比较，哪个小就输出相应的
  16.         字符值，同时对应的下标++，直到输出N个字符为止。  
  17.         
  18.   p.s. 
  19.         注意题目要求每一行输出80个字符。 
  20. */
  21. #include <cstdio>
  22. #include <cstring>
  23. #include <algorithm>
  25. using namespace std;
  26. const int MAX = 30000;
  28. struct Sfx{
  29.     int idx;
  30.     int key[2];
  31.     bool operator<(const Sfx &other)const{
  32.         return key[0] < other.key[0] || 
  33.             key[0] == other.key[0] && key[1] < other.key[1];    
  34.     }    
  35. };
  37. char cow[MAX * 2];
  38. Sfx sa[MAX * 2], temp_sa[MAX * 2];
  39. int rank[MAX * 2];
  40. int cnt[MAX * 2];
  42. void cSort(Sfx *in, int key, int len, Sfx *out){
  43.     memset(cnt, 0, sizeof(cnt));
  44.     for(int i = 0; i < len; i++)cnt[ in[i].key[key] ]++;
  45.     for(int i = 1; i < len; i++)cnt[i] += cnt[i - 1];
  46.     for(int i = len - 1; i >= 0; i--)
  47.         out[ --cnt[in[i].key[key]] ] = in[i];
  48. }
  50. void build_sfx(int len){
  51.     for(int i = 0; i < len; i++){
  52.         sa[i].idx = i;
  53.         sa[i].key[0] = cow[i];
  54.         sa[i].key[1] = i;
  55.     }
  56.     sort(sa, sa + len);
  57.     
  58.     for(int i = 0; i < len; i++)sa[i].key[1] = 0;
  59.     
  60.     int wid = 1;
  61.     while(wid < len){
  62.         rank[sa[0].idx] = 0;
  63.         for(int i = 1; i < len; i++){
  64.             rank[sa[i].idx] = rank[sa[i-1].idx];   
  65.             if(sa[i-1] < sa[i])rank[sa[i].idx]++;
  66.         }
  67.         for(int i = 0; i < len; i++){
  68.             sa[i].idx = i;
  69.             sa[i].key[0] = rank[i];
  70.             sa[i].key[1] = i + wid < len ? rank[i + wid] : 0;    
  71.         }
  72.         cSort(sa, 1, len, temp_sa); cSort(temp_sa, 0, len, sa);
  73.         wid *= 2;
  74.     }
  75. }
  77. int main(){
  78.     freopen("in.txt", "r", stdin);
  79.     freopen("out.txt", "w", stdout);
  80.     int N;
  81.     while(scanf("%d", &N) != EOF){
  82.         int i, j, count;
  83.         for(i = 0; i < N; i++)scanf("%s", cow+i);
  84.         for(i = 0; i < N; i++)cow[2 * N - 1 - i] = cow[i];
  85.         build_sfx(N * 2);
  86.             
  87.         for(i = 0, j = N, count = 0; i + j < 2 * N - 1; count++){
  88.             if(rank[i] < rank[j]){
  89.                 printf("%c", cow[i]);
  90.                 i++;
  91.             }
  92.             else{
  93.                 printf("%c", cow[j]);
  94.                 j++;
  95.             }
  96.             if(count % 80 == 79)printf("\n");
  97.         }
  98.         printf("%c\n", cow[i]);
  99.     }
  100.     return 0;    
  101. }



贪心法：
  1. /*
  2.   Title: Best Cow Line
  3.   Author: Jeff
  4.   Time:  2008/10/30
  5.   Complexity: O(N^2);
  6.   Result: 344K  2235MS/ AC
  7.   Reference: 
  8.         
  9.   Description: 
  10.         给定一个字符数组，可以从数组的开头或者结尾取出元素，
  11.         按取出顺序排成一列，使得他的字典序最小。
  12.   Tips:
  13.         贪心 
  14.   p.s. 
  15. */
  16. #include <cstdio>
  17. #include <cstring>
  19. using namespace std;
  20. const int MAX = 30000;
  22. char cow[MAX+1];
  24. int main(){
  25.     freopen("in.txt", "r", stdin);
  26.     freopen("out.txt", "w", stdout);
  27.     int N;
  28.     while(scanf("%d", &N) != EOF){
  29.         int i, j, count;
  30.         for(i = 0; i < N; i++)scanf("%s", cow+i);
  31.         for(i = 0, j = N-1, count = 0; i < j; count++){
  32.             int ii, jj;
  33.             for(ii = i, jj = j; ii <= j; ii++, jj--){
  34.                 if(cow[ii] > cow[jj]){
  35.                     printf("%c", cow[j]);
  36.                     j--;
  37.                     break;
  38.                 }
  39.                 else if(cow[ii] < cow[jj]){
  40.                     printf("%c", cow[i]);
  41.                     i++;
  42.                     break;    
  43.                 }
  44.             }
  45.             if(ii > j){
  46.                 printf("%c", cow[i]);    
  47.                 i++;
  48.             }
  49.             if(count % 80 == 79)printf("\n");
  50.         }
  51.         printf("%c\n", cow[i]);
  52.     }
  53.     return 0;    
  54. }
